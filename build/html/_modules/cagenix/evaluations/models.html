from cagenix import db



[docs]class Evaluation(db.Model):
    """The Evaluation model defines the questionaires that the Practitioners
    give to patients.

    :param db.Model: The SQLAlchemy base database model
    :type db.Model: object
    """
    __tablename__ = 'cagenix_evaluation'
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(255))
    active = db.Column(db.Boolean)
    questions = db.relationship('Question', backref='evaulation', lazy='dynamic')

    def __repr__(self):
        """__repr__ specifies what a Evaluation looks like when viewed in the shell

        :param self: An Evaluation Instance
        :type cls: class
        """
        return '<Evaluation %r>' % (self.name)

[docs]class Question(db.Model):
    """The Question model defines the questions that are in the questionaires.

    :param db.Model: The SQLAlchemy base database model
    :type db.Model: object
    """
    __tablename__ = 'cagenix_question'
    id = db.Column(db.Integer, primary_key=True)
    active = db.Column(db.Boolean)
    text = db.Column(db.String(255))
    order = db.Column(db.Integer)
    type = db.Column(db.String(255))
    answers = db.relationship('Answer', backref='question', lazy='dynamic')
    evaluation_id = db.Column(db.Integer, db.ForeignKey('cagenix_evaluation.id'), nullable=True)

    def __repr__(self):
        """__repr__ specifies what a question looks like when viewed in the shell

        :param self: A Question Instance
        :type cls: class
        """
        return '<Question %r>' % (self.text)
